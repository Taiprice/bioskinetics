function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var productLabel = (data.productName || '') + (data.variant ? ' — ' + data.variant : ''); var ingredientsText = data.ingredients || ''; var ingredientsList = ingredientsText.split(',').map(function(i){ return i.trim(); }).filter(function(i){ return i.length; }); var addressText = data.address || ''; var addressList = addressText.split(',').map(function(i){ return i.trim(); }).filter(function(i){ return i.length; }); // Column order: B Timestamp | C Email | D Address | E Product | F Ingredients | G Scents | H Size | I Color | J Price var row = [ new Date(), data.email || '', addressText, // keep the cell itself short/collapsed productLabel, ingredientsList.join(', '), // keep the cell itself short/collapsed data.scent || 'Unscented', data.size || '', data.color || '', data.amount || '' ]; var nextRow = sheet.getLastRow() + 1; sheet.getRange(nextRow, 2, 1, row.length).setValues([row]); // column 2 = B if (addressList.length) { sheet.getRange(nextRow, 4).setNote(addressList.join('\n')); // column 4 = D; hover/tap to expand into street/city/state zip } sheet.getRange(nextRow, 6).setNote(ingredientsList.join('\n')); // column 6 = F; hover/tap to expand into a vertical list // Text notification via T-Mobile email-to-SMS gateway try { var smsBody = 'New Bioskinetics order: ' + productLabel + ' - $' + (data.amount || '') + (data.email ? ' (' + data.email + ')' : ''); MailApp.sendEmail('7205603040@tmomail.net', 'Bioskinetics Order', smsBody); } catch (err) { MailApp.sendEmail('taiprice@bioskinetics.com', 'SMS notification failed', String(err)); } return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); }
